117    pub fn add_state<S: 'static + States + Clone + PartialEq>(&mut self, state: S) {
118        self.add_resource(NextState::<S>::Unchanged);
119        self.add_resource(State(state));
120
121        self.add_event::<StateTransitionEvent<S>>();
122
123        self.registered_states
124            .as_mut()
125            .unwrap()
126            .push(RegisteredState {
127                id: TypeId::of::<State<S>>(),
128                event_id: TypeId::of::<EventWriter<StateTransitionEvent<S>>>(),
129                update: |rec, id, event_id| {
130                    let mut state = rec.get(&id).unwrap().borrow_mut();
131                    let state = state.downcast_mut::<State<S>>().unwrap();
132
133                    let mut next_state =
134                        rec.get(&TypeId::of::<NextState<S>>()).unwrap().borrow_mut();
135                    let next_state = next_state.downcast_mut::<NextState<S>>().unwrap();
136
137                    let mut event_writer = rec.get(&event_id).unwrap().borrow_mut();
138                    let event_writer = event_writer
139                        .downcast_mut::<EventWriter<StateTransitionEvent<S>>>()
140                        .unwrap();
141
142                    if let Some(next_state) = next_state.take_next_state() {
143                        let last = state.get().clone();
144                        state.0 = next_state.clone();
145
146                        event_writer.send(StateTransitionEvent {
147                            from: last,
148                            to: next_state.clone(),
149                        });
150                    }
151                },
152            });
153
154        self.registered_transitions
155            .as_mut()
156            .unwrap()
157            .push(RegisteredTransition {
158                id: TypeId::of::<EventReader<StateTransitionEvent<S>>>(),
159                update: |rec, id| {
160                    let event_reader = rec.get(&id).unwrap().borrow();
161                    let event_reader = event_reader
162                        .downcast_ref::<EventReader<StateTransitionEvent<S>>>()
163                        .unwrap();
164
165                    for event in event_reader.queue.iter() {
166                        event.apply_transition(&rec);
167                    }
168                },
169            });
170    }

176    pub fn add_event<E: 'static + Event>(&mut self) {
177        self.add_resource(EventWriter::<E> { queue: vec![] });
178        self.add_resource(EventReader::<E> { queue: vec![] });
179
180        self.registered_events
181            .as_mut()
182            .unwrap()
183            .push(RegisteredEvent {
184                id: TypeId::of::<EventWriter<E>>(),
185                update: |rec, id| {
186                    let mut writer = rec.get(&id).unwrap().borrow_mut();
187                    let writer = writer.downcast_mut::<EventWriter<E>>().unwrap();
188
189                    writer.send_to_reader(&rec);
190                },
191            });
192    }

194    pub fn build_with_clock<const CLOCK: u32>(&mut self) -> TaskRunner<CLOCK> {
195        for task in self.startup_tasks.iter_mut() {
196            task.invoke(&self.resources.as_ref().unwrap());
197        }
198
199        let clock = SystemClock::default();
200        let start = clock.try_now().expect("Error can't start clock");
201        self.add_resource(clock);
202
203        TaskRunner {
204            tasks: self.update_tasks.take().unwrap(),
205            resources: self.resources.take().unwrap(),
206
207            registered_events: self.registered_events.take().unwrap(),
208            registered_states: self.registered_states.take().unwrap(),
209            registered_transitions: self.registered_transitions.take().unwrap(),
210
211            start_timestamp: start,
212        }
213    }

215    /// Builds a `TaskRunner` from the scheduler by invoking all startup tasks and preparing update tasks.
216    ///
217    /// # Parameters
218    ///
219    /// * `self` - A mutable reference to the `Scheduler` instance.
220    ///
221    /// # Returns
222    ///
223    /// * A `TaskRunner` that can execute scheduled tasks based on their timing.

230pub struct TaskRunner<const CLOCK: u32> {
231    tasks: Vec<(Microseconds<u64>, Microseconds<u64>, StoredTask)>,
232    resources: HashMap<TypeId, RefCell<Box<dyn Any>>, MAX_RESOURCES>,
233
234    start_timestamp: Instant<SystemClock<CLOCK>>,
235
236    registered_events: Vec<RegisteredEvent>,
237    registered_states: Vec<RegisteredState>,
238    registered_transitions: Vec<RegisteredTransition>,
239}
240
241impl<const CLOCK: u32> TaskRunner<CLOCK> {
242    /// Runs the scheduled tasks in a loop.
243    ///
244    /// This function continuously checks and runs tasks based on their scheduling criteria:
245    /// - It calculates the current time since the UNIX epoch in milliseconds.
246    /// - For each task, it checks if the elapsed time since the last execution is greater than the task's specified interval.
247    /// - If the condition is met, it invokes the task with the available resources and updates the last executed time to the current time.
248    pub fn run(&mut self) -> ! {
249        loop {
250            self.run_once();
251        }
252    }
253
254    pub fn run_once(&mut self) {
255        // Update the states
256        for state in self.registered_states.iter() {
257            (state.update)(&self.resources, state.id, state.event_id);
258        }
259
260        // Update the events
261        for event in self.registered_events.iter() {
262            (event.update)(&self.resources, event.id);
263        }
264
265        // Update the transitions
266        for transition in self.registered_transitions.iter() {
267            (transition.update)(&self.resources, transition.id);
268        }
269
270        let t = self
271            .resources
272            .get(&TypeId::of::<SystemClock<CLOCK>>())
273            .unwrap()
274            .borrow();
275
276        let clock = t.downcast_ref::<SystemClock<CLOCK>>().unwrap();
277
278        // Get the current time
279        let time = (clock.try_now().expect("Error getting current time") - self.start_timestamp)
280            .try_into()
281            .expect("Failed to convert time");
282
283        drop(t);
284
285        // Run the tasks
286        for task in self.tasks.iter_mut() {
287            if time - task.0 >= task.1 {
288                task.2.invoke(&mut self.resources);
289
290                task.0 = time;
291            }
292        }
293    }
