1use core::any::TypeId;
2use alloc::boxed::Box;
3use crate::scheduler::{Scheduler};
4use crate::states::{States, transition::Transition};

9use alloc::vec;

11pub struct OnEnter<S: States>(pub S);
12
13impl<I, T: Task + 'static, S: States + 'static + PartialEq + Clone> Schedule<I, T> for OnEnter<S> {
14    fn schedule_task(&self, s: &mut Scheduler, task: impl IntoTask<I, Task = T>) {
15        let t = s
16            .resources
17            .as_mut()
18            .unwrap()
19            .get_mut(&TypeId::of::<Transition<S>>());
20
21        let t = if t.is_none() {
22            s.add_resource(Transition::<S> {
23                on_enter: vec![],
24                on_exit: vec![],
25                on_transition: vec![],
26            });
27
28            s.resources
29                .as_mut()
30                .unwrap()
31                .get_mut(&TypeId::of::<Transition<S>>())
32        } else {
33            t
34        };
35
36        let arr = &mut t.unwrap().borrow_mut();
37        let arr = arr.downcast_mut::<Transition<S>>().unwrap();
38        let arr = &mut arr.on_enter;
39
40        // check if the state is already in the list
41        if let Some((_, tasks)) = arr.iter_mut().find(|(state, _)| state == &self.0) {
42            tasks.push(Box::new(task.into_task()));
43        } else {
44            arr.push((self.0.clone(), vec![Box::new(task.into_task())]));
45        }
46    }
47}
